Q2.array to print element in reverse
#include<iostream>
using namespace std;
int main()
{

    int arr[10];
    cout<<"enter 10 elements in array"<<endl;
    for(int i=0;i<10;i++)
    {
    cin>>arr[i];
    }
    cout<<"the array elements in reverse order are";
    for(int i=10;i>=0;i--)
    {
        cout<<arr[i]<<endl;
    }
    return 0;
}
